SELECT
    t.name,
    SUM(p.rows) rows
FROM sys.tables t
JOIN sys.partitions p
ON t.object_id=p.object_id
WHERE p.index_id IN (0,1)
GROUP BY
t.name
HAVING
SUM(p.rows)>100000
ORDER BY
rows DESC;
